<script>
/* ================= MOBILE SEARCH ================= */

const openMobileSearch = document.getElementById("openMobileSearch");
const closeMobileSearch = document.getElementById("closeMobileSearch");
const mobileSearch = document.getElementById("mobileSearch");
const mobileSearchInput = document.getElementById("mobileSearchInput");
const resortCards = document.querySelectorAll(".resort-card");
const resortSection = document.getElementById("services");

/* OPEN SEARCH */
openMobileSearch.onclick = () => {
  mobileSearch.style.display = "block";
  mobileSearchInput.focus();
};

/* CLOSE SEARCH */
closeMobileSearch.onclick = () => {
  mobileSearch.style.display = "none";
  mobileSearchInput.value = "";
  filterResorts("");
};

/* SCROLL ON FOCUS */
mobileSearchInput.addEventListener("focus", () => {
  resortSection.scrollIntoView({ behavior: "smooth" });
});

/* FILTER FUNCTION */
function filterResorts(query){
  query = query.toLowerCase();
  resortCards.forEach(card=>{
    const text =
      card.dataset.title.toLowerCase() +
      card.dataset.location.toLowerCase() +
      card.dataset.desc.toLowerCase() +
      card.innerText.toLowerCase();

    card.style.display = text.includes(query) ? "block" : "none";
  });
}

/* SEARCH WHILE TYPING */
mobileSearchInput.addEventListener("input", e=>{
  filterResorts(e.target.value);
});
</script>
